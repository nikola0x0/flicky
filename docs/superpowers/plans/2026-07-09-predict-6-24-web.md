```ts
const POS_INF_TICK = (1n << 30n) - 1n
export function deriveTicks(strike: bigint, isUp: boolean, tickSize: bigint) {
  const strikeTick = strike / tickSize
  return isUp
    ? { lowerTick: strikeTick, higherTick: POS_INF_TICK }   // UP = (K, +inf]
    : { lowerTick: 0n, higherTick: strikeTick }             // DOWN = [0, K]
}
```

```ts
export function buildStakedSwipeTx(a: {...}): Transaction {
  const tx = new Transaction()
  const { lowerTick, higherTick } = deriveTicks(a.strike, a.isUp, a.tickSize)
  const auth = tx.moveCall({ target: `${DEEPBOOK.accountPackageId}::account::generate_auth` })
  const pricer = tx.moveCall({
    target: `${DEEPBOOK.deepbookPredictPackageId}::expiry_market::load_live_pricer`,
    arguments: [
      tx.object(a.marketId), tx.object(DEEPBOOK.protocolConfigId), tx.object(DEEPBOOK.oracleRegistryId),
      tx.object(DEEPBOOK.pythFeedId), tx.object(DEEPBOOK.bsSpotFeedId), tx.object(DEEPBOOK.bsForwardFeedId),
      tx.object(DEEPBOOK.bsSviFeedId), tx.object(CONFIG.CLOCK_ID),
    ],
  })
  const order = tx.moveCall({
    target: `${DEEPBOOK.deepbookPredictPackageId}::expiry_market::mint_exact_quantity`,
    arguments: [
      tx.object(a.marketId), tx.object(a.wrapperId), auth, tx.object(DEEPBOOK.protocolConfigId), pricer,
      tx.pure.u64(lowerTick), tx.pure.u64(higherTick), tx.pure.u64(a.quantity),
      tx.pure.u64(1_000_000_000n),          // leverage 1x
      tx.pure.u64(2n ** 64n - 1n),          // max_cost uncapped
      tx.pure.u64(2n ** 64n - 1n),          // max_probability uncapped
      tx.object(DEEPBOOK.accumulatorRootId), tx.object(CONFIG.CLOCK_ID),
    ],
  })
  // chained record_swipe via codegen (order is the u256 result)
  tx.add(duel.recordSwipe({
    package: CONFIG.packageId,
    arguments: [a.duelId, tx.pure.u64(a.cardIdx), tx.pure.bool(a.isUp), tx.pure.u64(a.quantity), order, tx.object(CONFIG.CLOCK_ID)],
    typeArguments: [CONFIG.stakeType], // or duel's coin type
  }))
  return tx
}
```
